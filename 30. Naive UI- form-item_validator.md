### 紀錄內容：

有時候不一定要驗證整個表單
只驗證一個item

```html
<template>
  <n-form-item label="FormItem" :rule="rule">
    <n-input v-model:value="value" />
  </n-form-item>
</template>

<script lang="ts" setup>
import { ref } from 'vue'

const message = '單獨驗證item'
const valueRef = ref(message)
const rule = {
value: valueRef,
  rule: {
  trigger: ['input', 'blur'],
  validator() {
    if (valueRef.value !== message) {
      return new Error(message)
    }
   }
}
</script>
```

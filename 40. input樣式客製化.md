### 紀錄內容：

因應需求客製化了一個input
稍微研究了一下樣式

以輸入值是否是英數為例

```html
<template>
	<input
		:value="modelValue"
		class="sample-input"
		:class="{ 'sample-input-error': isError }"
		@input="enOnly"
	/>
</template>

<script setup lang="ts">
const modelValue = ref('')
const isError = ref(false)

const enOnly = (event) => {
	const target = e.target as HTMLInputElement
	if (target.value.test(/[^A-Za-z0-9]/)) {
		isError.value = false
	} else {
		isError.value = true
	}
}
</script>

<style>
.sample-input {
// 原本input的樣式
  width: 100%;
  height: 30px;
  border-radius: 3px;
  border: 1px solid #0002;
  padding: 0 12px;
  
// input內的游標顏色
  caret-color: green;
}

// onfocus的樣式
.scan-input:focus-visible {
  outline: 1px solid green;
}

// 修改placeholder樣式
.sample-input::placeholder {
  color: gray;
}

// error時的樣式
.sample-input-error {
  border: 1px solid #d03050;
  caret-color: red;
}

// error時onfocus的樣式
.sample-input-error:focus-visible {
  border: none;
  outline: 1px solid #d03050;
}
</style>
```

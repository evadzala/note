### 紀錄內容：

清空驗證(restoreValidation)

```html
<template>
  <n-space vertical>
    <n-space>
      <n-button @click="validateAll">
        執行全部規則
      </n-button>
      <n-button @click="clear">
        清空驗證
      </n-button>
    </n-space>
    <n-form ref="formInstRef" :model="model" :rules="rules">
      <n-form-item label="最短長度為3" path="fieldA">
        <n-input v-model:value="model.fieldA" />
      </n-form-item>
      <n-form-item label="最短長度2" path="fieldB">
        <n-input v-model:value="model.fieldB" />
      </n-form-item>
    </n-form>
  </n-space>
</template>

<script lang="ts" setup>
import { FormInst } from 'naive-ui'
import { ref } from 'vue'

const formInstRef = ref<FormInst | null>(null)
const model = ref({fieldA: '',fieldB: ''})
const rules = {
        fieldA: {
          key: 'a',
          required: true,
          min: 3,
          message: '最短長度為3'
        },
        fieldB: {
          required: true,
          min: 2,
          message: '最短長度為2'
        }
      }

const clear = () => {
	formInstRef.value?.restoreValidation()
}

const validateAll = () => {
  formInstRef.value?.validate((errors) => {
    if (errors) {
      console.error(errors)
    }
  })
},
</script>
```

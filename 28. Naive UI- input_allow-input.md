### 紀錄內容：

在`n-input` 使用`allow-input` 限制輸入格式
例如:

```html
<template>
  <n-space vertical>
    <n-input
      type="text"
      :allow-input="onlyAllowNumber"
      placeholder="只能输入数字"
    />
</template>

<script lang="ts">
import { defineComponent } from 'vue'

export default defineComponent({
  setup() {
    return {
      onlyAllowNumber: (value: string) => !value || /^\d+$/.test(value),
    }
  }
})
</script>
```
